#include <iostream>
#include <fstream>
#include <string>

using namespace std;

int main() {
	float c = 0.0;
	int i, kostil = 48;
	string Text, y;
	ofstream file("text.txt");
	if (file.is_open()) {
		cout << "Enter letters" << endl;
		getline(cin, Text);
		file << Text;
		file.close();
	}
	ifstream filesec("text.txt");
	if (filesec.is_open()) {
		getline(filesec, y);
		for (i = 0; i < y.size(); i++) {
			if (y[i] >= '0' && y[i] <= '9') {
				c = c + (y[i]) - kostil;
			}
		}
	}
	cout << c << endl;

	return 0;
}
