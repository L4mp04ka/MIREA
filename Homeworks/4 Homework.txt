#include <iostream>

int convert(char x)
{
    switch (x)
    {
    case 'I':
        return 1;
    case 'V':
        return 5;
    case 'X':
        return 10;
    case 'L':
        return 50;
    case 'C':
        return 100;
    case 'D':
        return 500;
    case 'M':
        return 1000;
    }
    return 0;
}
 
int main()
{
    string s;
    cin >> s;
    int x = 0, c1 = 0;
    for (int i = s.length() - 1; i >= 0; i--)
    {
        int c = convert(s[i]);
        if (c >= c1)
            c1 = c;
        if (c == c1)
            x += c;
        else
            x -= c;
    }
    cout << x;
}
