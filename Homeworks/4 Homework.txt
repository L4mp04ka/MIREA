#include <iostream>
#include <cmath>
using namespace std;

int Plochad(double a, double b)
{
	return a * b;
}

int Plochad(double a, double b, double h)
{
	return (a * b * h)/2; 
}

int Plochad(double r)
{
	return 3.14 * (r * r);
}



int main()
{
	int i = 0;
	double a, b, h, r;
	if (i == 0) {
		cout << "Enter a and b (S = a * b)" << endl;
		cin >> a >> b;
		float Pram = Plochad(a, b);
		i += 1;
		cout << Pram << " rectangle" << endl;
	}

	if (i == 1) {
		cout << "Enter a, b, h (S = (a * b * h)/2)" << endl;
		cin >> a >> b >> h;
		float Treug = Plochad(a, b, h);
		i += 1;
		cout << Treug << " triangle" << endl;
	}
	if (i == 2) {
		cout << "Enter r (S = Pi * r^2)" << endl;
		cin >> r;
		float Krug = Plochad(r);
		i += 1;
		cout << Krug << " circle" << endl;
	}
}
