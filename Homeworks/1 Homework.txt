#include <iostream>
#include <cmath>
using namespace std;
int main() {

	float R, r, h, V, L, S;
	cout << "Enter Radius: " << endl;
	cin >> R;
	cout << "Enter r: " << endl;
	cin >> r;
	cout << "Enter height: " << endl;
	cin >> h;
	if (R == 0 || r == 0 || h == 0 || R < r) {
		cout << "No answer";
	}
	else{
		if (((R - r) * (R - r) + h * h) < 0) {
			cout << "No answer";
		}
		else {
			V = (1.0 / 3.0 * 3.14 * h) * (R * R + R * r + r * r);
			cout << "V = " << V << endl;
			L = sqrt((R - r) * (R - r) + h * h);
			cout << "L = " << L << endl;
			S = 3.14 * (R * R + (R + r) * L + r * r);
			cout << "S = " << S << endl;
			return 0;
		}
	}
}
