#include <cmath>
using namespace std;

int main() {
	float S, p, n, r, m, k, l;
	cout << "Enter S" << endl;
	cin >> S;
	cout << "Enter p" << endl;
	cin >> p;
	cout << "Enter n" << endl;
	cin >> n;

	r = p / 100;
	l = 1 + r;
	k = pow(l, n);

	if (r > 0 && S > 0 && p > 0 && n > 0) {
		m = (S * r * k) / (12 * (k - 1));
		cout << "Answer is " << m << endl;
	}

	else {
		cout << "No answer" << endl;
	}

	return 0;
}

#include <iostream>
#include <cmath>
using namespace  std;

int main(int argc, char* argv[]) {
    float S, m;
    int n;

    cout << "Enter S, m,  n: " << endl;
    cin >> S >> m >> n;

    if (S <= 0 || m <= 0) {
        cout << "error" << endl;
        return 1;
    }

    if (n < 0) {
        cout << "error" << endl;
        return 1;
    }

    float r = 0;
    float step = 100;
    float prev_r = 0;

    if (m * 12 * n < S) {
        cout << "error" << endl;
        return 1;
    }

    while (step != 0.00001) {
        float current_m = (S * r * pow(1 + r, n)) / (12 * (pow(1 + r, n) - 1));
        if (current_m < m) {
            prev_r = r;
            r += step;
        }
        else if ((abs(m - current_m)) < 0.1) {
            cout << "p:  " << r * 100 << endl;
            return 0;
        }
        else {
            step /= 10;
            r = prev_r + step;
        }
    }
    cout << "error" << endl;
}
