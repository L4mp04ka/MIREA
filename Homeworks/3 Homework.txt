#include <iostream>
#include <cmath>
using namespace std;
int main() {

	float a, b, c, D, answer;
	cout << "Enter a" << endl;
	cin >> a;
	cout << "Enter b" << endl;
	cin >> b;
	cout << "Enter c" << endl;
	cin >> c;

	if (2 * a == 0) {
		cout << "No answer";
	}
	else{
	D = (b * b) - (4 * a * c);
	if (D == 0) {
		answer = (-b) / (2 * a);
		cout << "x = " << answer << endl;
	}
	else if (D > 0) {
		answer = ((-b) + sqrt(D)) / (2 * a);
		cout << "x1 = " << answer << endl;
		answer = ((-b) - sqrt(D)) / (2 * a);
		cout << "x2 = " << answer << endl;
	}
	else {
		cout << "No answer";
	}
}
	return 0;
}
